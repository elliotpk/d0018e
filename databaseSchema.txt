CREATE TABLE `user`(
	`id` INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
	`email` VARCHAR(45) NOT NULL,
	`hashed_pass` VARCHAR(255) NOT NULL,
	`address` VARCHAR(255) NOT NULL DEFAULT '',
	`billing` VARCHAR(255) NOT NULL DEFAULT '',
	`isDeleted` TINYINT NOT NULL DEFAULT '0',
	`user_type` CHAR(1) NOT NULL DEFAULT 'U'
);

CREATE TABLE `item`(
	`id` INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
	`name` VARCHAR(255) NOT NULL,
	`image` VARCHAR(255) DEFAULT NULL,
	`price` INT NOT NULL
);

CREATE TABLE `listing`(
	`id` INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
	`active` TINYINT NOT NULL,
	`item:id` INT UNSIGNED NOT NULL,
	`date` VARCHAR(45) NOT NULL,
	FOREIGN KEY (`item:id`)
	REFERENCES item(`id`)
		ON DELETE CASCADE
);

CREATE TABLE `order`(
	`id` INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
	`date` VARCHAR(45) NOT NULL,
	`user:id` INT UNSIGNED NOT NULL,
	`sum` INT NOT NULL,
	FOREIGN KEY (`user:id`)
	REFERENCES user(`id`)
		ON DELETE CASCADE
);

CREATE TABLE `comment`(
	`id` INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
	`text` VARCHAR(300) NOT NULL,
	`user:id` INT UNSIGNED NOT NULL,
	`rating` INT NOT NULL,
	`isDeleted` TINYINT NOT NULL DEFAULT '0',
	`listing:id` INT UNSIGNED NOT NULL,
	FOREIGN KEY (`user:id`)
	REFERENCES user(`id`)
		ON DELETE CASCADE,
	FOREIGN KEY (`listing:id`)
	REFERENCES listing(`id`)
		ON DELETE CASCADE
);

CREATE TABLE `attributes`(
	`id` INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
	`name` VARCHAR(45) NOT NULL
);

CREATE TABLE `attribute_value`(
	`item:id` INT UNSIGNED NOT NULL,
	`attributes:id` INT UNSIGNED NOT NULL,
	`value` VARCHAR(45) NOT NULL,
	PRIMARY KEY (`item:id`, `attributes:id`),
	FOREIGN KEY (`item:id`)
	REFERENCES item(`id`)
		ON DELETE CASCADE,
	FOREIGN KEY (`attributes:id`)
	REFERENCES attributes(`id`)
		ON DELETE CASCADE
);

CREATE TABLE `cart`(
	`id` INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
	`user:id` INT UNSIGNED NOT NULL,
	`item:id` INT UNSIGNED NOT NULL,
	`available` TINYINT NOT NULL,
	FOREIGN KEY (`user:id`)
	REFERENCES user(`id`)
		ON DELETE CASCADE,
	FOREIGN KEY (`item:id`)
	REFERENCES item(`id`)
		ON DELETE CASCADE
);

CREATE TABLE `order_items`(
	`id` INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
	`item:id` INT UNSIGNED NOT NULL,
	`order:id` INT UNSIGNED NOT NULL,
	FOREIGN KEY (`item:id`)
	REFERENCES item(`id`)
		ON DELETE CASCADE,
	FOREIGN KEY (`order:id`)
	REFERENCES `order`(`id`)
		ON DELETE CASCADE
);
